阿里云静态云存储接口类

功能说明
阿里云静态云存储文件上传、删除接口

部署说明

下载 alioos.zip 解压后部署到 phpGrace\models 文件夹下


配置说明
打开 phpGrace\models\alioos.php，正确填写以下配置：

//阿里 OOS 配置
define('OSS_ACCESS_ID',  '******'); //阿里云账户中心获取
define('OSS_ACCESS_KEY', '******'); //阿里云账户中心获取
define('OSS_HOST',       'http://oss-cn-beijing.aliyuncs.com/');//OOS主机地址OOS中心获取
define('OSS_BUCKET',     '******'); //空间名称，由开发者创建


使用演示

<?php
class indexController extends grace{
    
    //上传到OOS
    public function index(){
        $oos = model('alioos');
	$res = $oos->toOos('./1.jpg', '1.jpg');
    }
    
    //从 OOS 中删除文件
    public function delete(){
        $oos = model('alioos');
        $oos->remove('1.jpg');
    }
}
